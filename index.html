Base de datos en la nube

Cómo funciona: El CRM guarda datos localmente. Conecta Sheets para sincronizar con un clic y acceder desde cualquier dispositivo.
Paso 1 — Crea la hoja
1sheets.google.com → hoja nueva → llámala "Mi CRM"
Cabeceras en la fila 1 desde A1:
id nombre tel email tipo estado precio cobrado fecha notas
2Copia el ID de la URL de tu hoja
URL: spreadsheets/d/ID_AQUI/edit
Paso 2 — Apps Script
3Extensiones → Apps Script → borra todo y pega:
const SHEET_ID = 'PEGA_TU_ID_AQUI'; const SHEET_NAME = 'Hoja 1'; function doGet(e) { try { const sheet = SpreadsheetApp .openById(SHEET_ID).getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return out([]); const h = data[0]; return out(data.slice(1).map(r => { const o = {}; h.forEach((k,i) => o[k] = r[i]); return o; })); } catch(e) { return out({error: e.toString()}); } } function doPost(e) { try { const sheet = SpreadsheetApp .openById(SHEET_ID).getSheetByName(SHEET_NAME); let p; if (e.parameter && e.parameter.payload) { p = JSON.parse(e.parameter.payload); } else { p = JSON.parse(e.postData.contents); } if (p.action === 'sync') { sheet.clearContents(); sheet.appendRow(['id','nombre','tel','email', 'tipo','estado','precio','cobrado','fecha','notas']); p.data.forEach(c => sheet.appendRow([ c.id, c.nombre, c.tel||'', c.email||'', c.tipo, c.estado, c.precio, c.cobrado, c.fecha||'', c.notas||'' ])); return out({ok: true}); } } catch(e) { return out({error: e.toString()}); } } function out(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
